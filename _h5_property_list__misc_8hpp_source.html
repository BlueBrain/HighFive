19 inline hid_t convert_plist_type(PropertyType propertyType) {
20  // The HP5_XXX are macros with function calls so we can't assign
21  // them as the enum values
22  switch (propertyType) {
23  case PropertyType::OBJECT_CREATE:
24  return H5P_OBJECT_CREATE;
25  case PropertyType::FILE_CREATE:
26  return H5P_FILE_CREATE;
27  case PropertyType::FILE_ACCESS:
28  return H5P_FILE_ACCESS;
29  case PropertyType::DATASET_CREATE:
30  return H5P_DATASET_CREATE;
31  case PropertyType::DATASET_ACCESS:
32  return H5P_DATASET_ACCESS;
33  case PropertyType::DATASET_XFER:
34  return H5P_DATASET_XFER;
35  case PropertyType::GROUP_CREATE:
36  return H5P_GROUP_CREATE;
37  case PropertyType::GROUP_ACCESS:
38  return H5P_GROUP_ACCESS;
39  case PropertyType::DATATYPE_CREATE:
40  return H5P_DATATYPE_CREATE;
41  case PropertyType::DATATYPE_ACCESS:
42  return H5P_DATATYPE_ACCESS;
43  case PropertyType::STRING_CREATE:
44  return H5P_STRING_CREATE;
45  case PropertyType::ATTRIBUTE_CREATE:
46  return H5P_ATTRIBUTE_CREATE;
47  case PropertyType::OBJECT_COPY:
48  return H5P_OBJECT_COPY;
49  case PropertyType::LINK_CREATE:
50  return H5P_LINK_CREATE;
51  case PropertyType::LINK_ACCESS:
52  return H5P_LINK_ACCESS;
53  default:
54  HDF5ErrMapper::ToException<PropertyException>(
55  "Unsupported property list type");
56  }

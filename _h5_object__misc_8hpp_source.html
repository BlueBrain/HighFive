73static inline ObjectType _convert_object_type(const H5I_type_t& h5type) {
74 switch (h5type) {
75 case H5I_FILE:
76 return ObjectType::File;
77 case H5I_GROUP:
78 return ObjectType::Group;
79 case H5I_DATATYPE:
80 return ObjectType::UserDataType;
81 case H5I_DATASPACE:
82 return ObjectType::DataSpace;
83 case H5I_DATASET:
84 return ObjectType::Dataset;
85 case H5I_ATTR:
86 return ObjectType::Attribute;
87 default:
88 return ObjectType::Other;
89 }
90}

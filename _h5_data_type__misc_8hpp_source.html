257 // Calculate the padding required to align an element of a struct
258 #define _H5_STRUCT_PADDING(current_size, member_size) (((member_size) - (current_size)) % (member_size))
259 
260 inline void CompoundType::create(size_t size) {
261  if (size == 0) {
262  size_t current_size = 0, max_type_size = 0;
263 
264  // Do a first pass to find the total size of the compound datatype
265  for (auto& member: members) {
266  size_t member_size = H5Tget_size(member.base_type.getId());
267  if (member_size == 0) {
268  throw DataTypeException("Cannot get size of DataType with hid: " +
269  std::to_string(member.base_type.getId()));
270  }
271 
272  // Set the offset of this member within the struct according to the
273  // standard alignment rules
274  member.offset = current_size + _H5_STRUCT_PADDING(current_size, member_size);
275 
276  // Set the current size to the end of the new member
277  current_size = member.offset + member_size;
278 
279  max_type_size = std::max(max_type_size, member_size);
280  }
281 
282  size = current_size + _H5_STRUCT_PADDING(current_size, max_type_size);
283  }
284 
285  // Create the HDF5 type
286  if((_hid = H5Tcreate(H5T_COMPOUND, size)) < 0) {
287  HDF5ErrMapper::ToException<DataTypeException>(
288  "Could not create new compound datatype");
289  }
290 
291  // Loop over all the members and insert them into the datatype
292  for (const auto& member: members) {
293  if(H5Tinsert(_hid, member.name.c_str(), member.offset, member.base_type.getId()) < 0) {
294  HDF5ErrMapper::ToException<DataTypeException>(
295  "Could not add new member to datatype"
296  );
297  }
298  }
299 }

416 template <typename T>
417 inline DataType create_and_check_datatype() {
418 
419  DataType t = create_datatype<T>();
420  // Skip check if the base type is a variable length string
421  if (t.isVariableStr()) {
422  return t;
423  }
424  // Check that the size of the template type matches the size that HDF5 is
425  // expecting.
426  if(!t.isReference() && (sizeof(T) != t.getSize())) {
427  std::ostringstream ss;
428  ss << "Size of array type " << sizeof(T)
429  << " != that of memory datatype " << t.getSize()
430  << std::endl;
431  throw DataTypeException(ss.str());
432  }
433 
434  return t;
435 }

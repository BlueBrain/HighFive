60 template <typename Derivate>
61 inline Selection
62 SliceTraits<Derivate>::select(const std::vector<size_t>& offset,
63  const std::vector<size_t>& count,
64  const std::vector<size_t>& stride) const {
65  // hsize_t type conversion
66  // TODO : normalize hsize_t type in HighFive namespace
67  std::vector<hsize_t> offset_local(offset.size());
68  std::vector<hsize_t> count_local(count.size());
69  std::vector<hsize_t> stride_local(stride.size());
70  std::copy(offset.begin(), offset.end(), offset_local.begin());
71  std::copy(count.begin(), count.end(), count_local.begin());
72  std::copy(stride.begin(), stride.end(), stride_local.begin());
73 
74  DataSpace space = static_cast<const Derivate*>(this)->getSpace().clone();
75  if (H5Sselect_hyperslab(space.getId(), H5S_SELECT_SET, offset_local.data(),
76  stride.empty() ? NULL : stride_local.data(),
77  count_local.data(), NULL) < 0) {
78  HDF5ErrMapper::ToException<DataSpaceException>(
79  "Unable to select hyperslap");
80  }
81 
82  return Selection(DataSpace(count), space,
83  details::get_dataset(static_cast<const Derivate*>(this)));
84 }

86 template <typename Derivate>
87 inline Selection
88 SliceTraits<Derivate>::select(const std::vector<size_t>& columns) const {
89 
90  const DataSpace& space = static_cast<const Derivate*>(this)->getSpace();
91  const DataSet& dataset =
92  details::get_dataset(static_cast<const Derivate*>(this));
93  std::vector<size_t> dims = space.getDimensions();
94  std::vector<hsize_t> counts(dims.size());
95  std::copy(dims.begin(), dims.end(), counts.begin());
96  counts[dims.size() - 1] = 1;
97  std::vector<hsize_t> offsets(dims.size(), 0);
98 
99  H5Sselect_none(space.getId());
100  for (std::vector<size_t>::const_iterator i = columns.begin();
101  i != columns.end(); ++i) {
102 
103  offsets[offsets.size() - 1] = *i;
104  if (H5Sselect_hyperslab(space.getId(), H5S_SELECT_OR, offsets.data(),
105  0, counts.data(), 0) < 0) {
106  HDF5ErrMapper::ToException<DataSpaceException>(
107  "Unable to select hyperslap");
108  }
109  }
110 
111  dims[dims.size() - 1] = columns.size();
112  return Selection(DataSpace(dims), space, dataset);
113 }

115 template <typename Derivate>
116 inline Selection
117 SliceTraits<Derivate>::select(const ElementSet& elements) const {
118  const hsize_t* data = NULL;
119  const std::size_t length = elements._ids.size();
120  std::vector<hsize_t> raw_elements;
121 
122  // optimised at compile time
123  // switch for data conversion on 32bits platforms
124  if (std::is_same<std::size_t, hsize_t>::value) {
125  // `if constexpr` can't be used, thus a reinterpret_cast is needed.
126  data = reinterpret_cast<const hsize_t*>(&(elements._ids[0]));
127  } else {
128  raw_elements.resize(length);
129  std::copy(elements._ids.begin(), elements._ids.end(),
130  raw_elements.begin());
131  data = &(raw_elements[0]);
132  }
133 
134  DataSpace space = static_cast<const Derivate*>(this)->getSpace().clone();
135  if (H5Sselect_elements(space.getId(), H5S_SELECT_SET, length, data) < 0) {
136  HDF5ErrMapper::ToException<DataSpaceException>(
137  "Unable to select elements");
138  }
139 
140  return Selection(DataSpace(length), space,
141  details::get_dataset(static_cast<const Derivate*>(this)));
142 }

144 template <typename Derivate>
145 template <typename T>
146 inline void SliceTraits<Derivate>::read(T& array) const {
147  typedef typename std::remove_const<T>::type type_no_const;
148 
149  type_no_const& nocv_array = const_cast<type_no_const&>(array);
150 
151  const size_t dim_array = details::array_dims<type_no_const>::value;
152  DataSpace space = static_cast<const Derivate*>(this)->getSpace();
153  DataSpace mem_space = static_cast<const Derivate*>(this)->getMemSpace();
154 
155  if (!details::checkDimensions(mem_space, dim_array)) {
156  std::ostringstream ss;
157  ss << "Impossible to read DataSet of dimensions "
158  << mem_space.getNumberDimensions() << " into arrays of dimensions "
159  << dim_array;
160  throw DataSpaceException(ss.str());
161  }
162 
163  // Create mem datatype
164  const AtomicType<typename details::type_of_array<type_no_const>::type>
165  array_datatype;
166 
167  // Apply pre-read conversions
168  details::data_converter<type_no_const> converter(nocv_array, mem_space);
169 
170  if (H5Dread(
171  details::get_dataset(static_cast<const Derivate*>(this)).getId(),
172  array_datatype.getId(),
173  details::get_memspace_id((static_cast<const Derivate*>(this))),
174  space.getId(), H5P_DEFAULT,
175  static_cast<void*>(converter.transform_read(nocv_array))) < 0) {
176  HDF5ErrMapper::ToException<DataSetException>(
177  "Error during HDF5 Read: ");
178  }
179 
180  // re-arrange results
181  converter.process_result(array);
182 }

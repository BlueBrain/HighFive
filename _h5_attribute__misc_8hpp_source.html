56 
57  type_no_const& nocv_array = const_cast<type_no_const&>(array);
58 
59  const size_t dim_array = details::array_dims<type_no_const>::value;
60  DataSpace space = getSpace();
61  DataSpace mem_space = getMemSpace();
62 
63  if (!details::checkDimensions(mem_space, dim_array)) {
64  std::ostringstream ss;
65  ss << "Impossible to read attribute of dimensions "
66  << mem_space.getNumberDimensions() << " into arrays of dimensions "
67  << dim_array;
68  throw DataSpaceException(ss.str());
69  }
70 
71  // Create mem datatype

73  array_datatype;
74 
75  // Apply pre read conversions
76  details::data_converter<type_no_const> converter(nocv_array, mem_space);
77 
78  if (H5Aread(getId(), array_datatype.getId(),
79  static_cast<void*>(converter.transform_read(nocv_array))) < 0) {
80  HDF5ErrMapper::ToException<AttributeException>(
81  "Error during HDF5 Read: ");

83
